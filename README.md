```R
plot_dist <- function(x, threshold, use_boxplot = TRUE) {
   if (length(x) > threshold & use_boxplot) {
       boxplot(x)
   } else if (length(x) > threshold & !use_boxplot) {
       hist(x)
   } else {
       stripchart(x)
   }
}
```
